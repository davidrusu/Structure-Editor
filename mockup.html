import Data.Char
filter :: ( a -> Bool ) -> [a] -> [a]
filter _ [] = []
filter f ( x : xs ) = case f x of
True -> x : ( filter f xs )
False -> filter f xs map :: ( a -> b ) -> [ a ] -> [b]
map _ [] = []
map f ( x : xs ) = f x : map f xs main :: IO ()
main = do input <- getLine
let upCase = map toUpper input
putStrLn upCase filter' :: ( a -> Bool ) -> [a] -> [a]
filter' f xs = foldr decider [] xs
where
decider x a = if f x then x : a else a
